    \pstree[linewidth=0.8pt,nodesepA=0pt]{\erlcirc{|}~[tnpos=l]{\(n\)}}{

            \TR{\(I_{n-1}\)}
            \erlnode{[]}
          }
        }
      }
      \psset{nodesepA=0pt,nodesepB=2pt,linestyle=solid,linewidth=0.8pt}
      \erlnode{[]}
    }
    \psset{nodesepA=0pt,nodesepB=2pt,linestyle=solid,linewidth=0.8pt}

  }
